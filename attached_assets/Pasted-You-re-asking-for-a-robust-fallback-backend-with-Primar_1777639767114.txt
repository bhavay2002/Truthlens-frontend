## 🔹 `server.js`

```js
import express from "express";
import axios from "axios";
import cors from "cors";
import dotenv from "dotenv";

dotenv.config();

const app = express();
app.use(cors());
app.use(express.json());

const MODEL_API = process.env.MODEL_API;
const GEMINI_API_KEY = process.env.GEMINI_API_KEY;

/* ----------------------------------------
   Gemini Call
-----------------------------------------*/
async function callGemini(text) {
  const url = `https://generativelanguage.googleapis.com/v1beta/models/gemini-pro:generateContent?key=${GEMINI_API_KEY}`;

  const prompt = `
Analyze this news text and classify:
- REAL or FAKE
- confidence (0 to 1)

Text:
"${text}"

Respond in JSON:
{
  "prediction": "REAL or FAKE",
  "confidence": number,
  "fake_probability": number
}
`;

  const res = await axios.post(url, {
    contents: [{ parts: [{ text: prompt }] }]
  });

  const raw = res.data.candidates?.[0]?.content?.parts?.[0]?.text;

  try {
    return JSON.parse(raw);
  } catch {
    // fallback parse
    return {
      prediction: raw.includes("FAKE") ? "FAKE" : "REAL",
      confidence: 0.7,
      fake_probability: raw.includes("FAKE") ? 0.7 : 0.3
    };
  }
}

/* ----------------------------------------
   Normalize Output (IMPORTANT)
-----------------------------------------*/
function formatResponse(base) {
  return {
    prediction: base.prediction || "REAL",
    fake_probability: base.fake_probability ?? 0.5,
    confidence: base.confidence ?? 0.5,

    // minimal fields to satisfy frontend
    emotion: {
      dominant_emotion: "neutral",
      emotion_scores: { neutral: 1 }
    },

    explainability: {
      lime: {
        structured: [
          { token: "sample", importance: 0.2 }
        ]
      }
    }
  };
}

/* ----------------------------------------
   MAIN ROUTE
-----------------------------------------*/
app.post("/analyze", async (req, res) => {
  const { text } = req.body;

  if (!text || text.length < 20) {
    return res.status(422).json({ error: "Text too short" });
  }

  try {
    // run both in parallel
    const modelPromise = axios.post(MODEL_API, { text })
      .then(r => ({ source: "model", data: r.data }))
      .catch(() => null);

    const geminiPromise = callGemini(text)
      .then(data => ({ source: "gemini", data }))
      .catch(() => null);

    // timeout after 30 sec
    const timeout = new Promise(resolve =>
      setTimeout(() => resolve("timeout"), 30000)
    );

    const result = await Promise.race([
      Promise.allSettled([modelPromise, geminiPromise]),
      timeout
    ]);

    let modelResult = null;
    let geminiResult = null;

    if (result !== "timeout") {
      const [m, g] = result;

      modelResult = m.status === "fulfilled" ? m.value : null;
      geminiResult = g.status === "fulfilled" ? g.value : null;
    }

    // DECISION LOGIC
    let final;

    if (modelResult && modelResult.data) {
      final = modelResult.data;
    } else if (geminiResult && geminiResult.data) {
      final = geminiResult.data;
    } else {
      final = {
        prediction: "REAL",
        confidence: 0.5,
        fake_probability: 0.5
      };
    }

    return res.json(formatResponse(final));

  } catch (err) {
    return res.status(500).json({ error: "Internal error" });
  }
});

/* ----------------------------------------
   HEALTH
-----------------------------------------*/
app.get("/health", (req, res) => {
  res.json({ status: "healthy" });
});

/* ----------------------------------------
   START
-----------------------------------------*/
app.listen(process.env.PORT, () => {
  console.log(`Server running on ${process.env.PORT}`);
});
```
